# 3. Stop service

# 4. Archive old version
mv ~/Chimera ~/Chimera_archive_$(date +%Y%m%d_%H%M%S)

# 5. Extract new version
tar xzf Chimera_v4.13.0.tar.gz

# 6. Compile (CRITICAL: clean build required)

# 7. Verify build success

# 8. Start service
sudo systemctl start chimera
